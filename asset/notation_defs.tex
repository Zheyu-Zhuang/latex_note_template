% Frame
\providecommand{\frameX}[1]{\{#1\}}

% Transformation 
% Usege: \tf[1][2]{3}[4][5]. If one optional argument appears on either side of var, it will be considered as the first argument on that side.
% #1 (optional): where the quantity is expressed w.r.t or the coordinates
% #2 (optional): where quantity is the measured w.r.t or reference.
% #3 (mendatory): the symbol of the transformation, typically R, t, X
% #4 (optional): the tip or index.
% #5 (optional): the operator or superscript - usually \top, or \vee, or \times, etc.

\NewDocumentCommand{\tf}{O{} O{} m o O{}}  
  {
    \tensor*[^{#1}_{#2}]{\mathbf{#3}}{_{#4}^{#5}}
  }
